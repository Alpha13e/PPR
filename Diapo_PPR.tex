\begin{frame}
    \frametitle{Introduction}
    Les distances stellaires étant sans commune mesure avec celles terrestres, nous devons inventer de nouvelles méthodes de mesure:
\\[0.5cm]
\begin{description}
    \item[La Parallaxe] calcule la distance d'un objet par trigonométrie en mesurant son angle d'observation.
    \item[Les chandelles standard] sont des objets dont une propriété particulière permet de déterminer sa luminosité réelle,
        et donc sa distance.
    \item[L'effet Doppler (ou redshift)] consiste à mesurer le décalage spectral d'un objet, qui est lié à sa vitesse relative par la loi de même nom. 
        Cette méthode ne sera pas traitée ici.
\end{description}
\end{frame}

\subsection{Application}
\begin{frame}
    \frametitle{La Parallaxe}
    \framesubtitle{Application}
    \begin{columns}
        \begin{column}{0.3\textwidth}
            \textbf{Source:}
            \\[8pt]
            \includegraphics{gaia_logo.png}

            Le satellite est positionné sur le point L2, mais les calculs sont ramenés pour $r=1$~ua.

            La définition du parsec étant fonction de l'arcseconde, $[d]=pc$.
        \end{column}
        \begin{column}{0.7\textwidth}
            \textbf{Erreur:}

            On a un calcul de type $d=\frac{cte}\varpi$ avec $\sigma_d$ l'incertitude sur la distance et celle sur la mesure $\sigma_\varpi$ fournie:
            \[
            d\, \varpi=cte \Leftrightarrow \mathrm{d} (d\,\varpi)=0 \Leftrightarrow d\,\mathrm{d} \varpi=-\varpi\,\mathrm{d} d
            \]
            \begin{equation*}\label{eq.parallax-error}
                \frac{\sigma_\varpi}{\varpi}=\left|\frac{\sigma_d}{d}\right| \Longrightarrow
                \sigma_d=d\left|\frac{\sigma_\varpi}{\varpi}\right|
            \end{equation*}
            Cependant, selon \citet[Section~3.2]{DR2-parallax}, deux conditions supplémentaires s'imposent:
            \begin{equation*}
                f=\frac{\sigma_\varpi}{\varpi}<20\%
            \end{equation*}
            \begin{equation*}
                \varpi>0
            \end{equation*}
            Note: On applique aussi la correction de $-0.029\,mas$ due aux mesures de Gaia.
        \end{column}
    \end{columns}
\end{frame}

\subsection{Programme}
\begin{frame}[fragile] % étonnament, permet de compiler
    \frametitle{La Parallaxe}
    \framesubtitle{Programme}
\begin{lstlisting}
# Importation des fichiers (coupee)
output.write("source_id\tparallax (mas)\tparallax_error(mas)\tdistance(pc)\tdistance_error (pc)\n")
# Automatisation
i=0 # compte les echecs
n=100 # nombre de lignes a lire, arbitraire

for _ in range(n):
    l=input.readline()
    id,par,perr,*_=l.split(',')

    par=float(par)+0.029 # erreur zero-point (-biais, biais=-0.029)
    f=abs(float(perr)/par) # calcul de la qualite
    dist=1/(par*1e-3) 

    if f<0.2 and dist>0:
        derr=dist*f # propagation de l'erreur
        output.write(f"{id}\t{par}\t{perr}\t{dist}\t{derr}\n")
    else:
        i+=1

output.close()
input.close()
print(f"Done. {i} errors over {n}: Success rate {1-i/n:.2f}")
\end{lstlisting}
\end{frame}

\subsection{Théorie}
\begin{frame}
    \frametitle{Chandelles standard: les Céphéides}
    \framesubtitle{Théorie}
    \textbf{La magnitude,} unité de luminosité sans dimension:
    \begin{block}{La magnitude absolue}
        \begin{equation*}
            M=-2.5\,\log_{10}(\frac{l}{4\pi(10\,pc)^2L_{Soleil}})+k
        \end{equation*}
    \end{block}

    \begin{block}{La magnitude apparente}
        \begin{equation*}\label{eq.module_distance}
            \mu = m-M = 5\log_{10}(d)-5 
            \Longleftrightarrow d = 10^{\frac{m-M+5}{5}}
        \end{equation*}
        Magnitude mesurée à une distance $d$, liée à $M$ par le \textbf{module de distance}.
    \end{block}
    Cependant, des poussières absorbent les rayonnements, c'est \textbf{l'extinction interstellaire}.
    Pour pallier ce problème on utilise l'indice de Wesenheit.
    \\[8pt]
    \textit{On note que plus un astre est lumineux, plus $M$ est négatif.}
\end{frame}

\subsection{Calibration}
\begin{frame}
    \frametitle{Chandelles standard: les Céphéides}
    \framesubtitle{Calibration}
    J'ai dû déléguer la calibration, utiliser celle de chercheurs car:
    \begin{itemize}
        \item Peu de céphéides sont suffisamment proches pour utiliser la parallaxe et fiables.
        \item Elles sont brillantes et de luminosité variable: on doit trouver un proxy.
        \item Les problèmes de la parallaxe s'appliquent toujours: on doit utiliser l'Astrometric Based Luminosity (ABL):
         \begin{equation*}\label{eq.ABL}
            \text{ABL} = 10^{0.2 M} = \varpi 10^{0.2 m - 2}
         \end{equation*}
    \end{itemize}
    On a utilisé la théorie de \citet{cepheid-calibration}, mais la calibration est liée à nos données \citet{Ripepi_datapaper},
    obtenues via la plateforme VizieR \textit{(Université de Strasbourg)}.
    %{\centering\includegraphics[height=0.1\textwidth]{vizier_logo.png}}
\end{frame}

\begin{frame}
    \frametitle{Chandelles standard: les Céphéides}
    \framesubtitle{Calibration}
    La fonction de Wesenheit adaptée aux capteurs de Gaia devient:
    \begin{equation*}\label{eq.Wesenheit}
        W(G,G_{BP},G_{RP})=m=G-1.90(G_{BP}-G_{RP})
    \end{equation*}

    On lit le tableau 3: Nous employons la méthode de l'ABL ($PW_A$) appliquée aux céphéides classiques (DCEP\_F)
    Alors, $\sigma_{ABL}=0.011$. On obtient la loi de Leavitt:
    \begin{equation*}\label{eq.leavitt-calibree}
        W_A=M=-2.837-3.183\log_{10}(P)
    \end{equation*}

    On mêle les 3 équations: %dfrac agrandit visuellement la fraction

    \[
        \boxed{ 
        d = 10^{\dfrac{G-1.90(G_{BP}-G_{RP})+2.837+3.183\log_{10}(P)+5}{5}}}
    \]
\end{frame}

\subsection{Programme}
\begin{frame}[fragile,allowframebreaks]
    \frametitle{Chandelles standard: les Céphéides}
    \framesubtitle{Programme}
    \begin{lstlisting}
from numpy import log10, log

n=2000
count=0
for _ in range(52): # la 53e est les premieres donnees
    input.readline()

output.write("source_id\tdistance_parallax(pc)\t"
    f"distance_error_parallax (pc)\tdistance_cepheid"
    f"\tdistance_error_cepheid\n")

for _ in range(n):
    line=input.readline().replace(' ','').strip().split(';')
    if line[16]=='DCEP_F':
        count+=1 # ouverture
        id=line[0]
        period,G,Gbp,Grp,parallax,errparallax=line[6:12]

        # Paralaxe
        parallax=float(parallax)+0.046 # celle adaptee selon ripepi p.4
        f=abs(float(errparallax)/parallax)
        pardist=1/(parallax*1e-3) 
        if f<0.2 and pardist>0:
            parderr=str(pardist*f)
            pardist=str(pardist)
        else:
            pardist='NULL'
            parderr='NULL'

        # Cepheide

        M=-2.837-3.183*log10(float(period))
        m=float(G)-1.9*(float(Gbp)-float(Grp))

        cepdist=10**((m-M+5)/5)
        ceperr=0.2*log(10)*cepdist*0.011
        
        output.write(f"{id}\t{pardist}\t{parderr}\t"
        f"{cepdist}\t{ceperr}\n")

input.close()
output.close()
print(count)
\end{lstlisting}
\end{frame}
